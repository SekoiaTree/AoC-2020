# AoC-2020

Advent of Code 2020 in Rust, Julia, Golang, Typescript, Kotlin, Swift, C, Nim, and Python

```ts
function writtenIn(day: number): string {
  const choices = [
    "Rust",
    "Golang",
    "Julia",
    "Typescript",
    "Python",
    "C",
    "Nim",
    "Kotlin",
    "Swift",
  ];
  return choices[(day - 1) % choices.length];
}
```
